1. SELECT * FROM directors WHERE nationality = 'American' ORDER BY date_of_birth;
2. SELECT DISTINCT nationality FROM directors;
3. SELECT first_name, last_name, date_of_birth from actors WHERE gender='F' ORDER BY date_of_birth LIMIT 10;
4. SELECT * FROM movie_revenues ORDER BY international_takings DESC LIMIT 3; 
5. SELECT CONCAT(first_name,' ',last_name) AS "full_name" FROM directors;
6. SELECT * FROM actors WHERE first_name IS NULL OR date_of_birth IS NULL;
7. SELECT COUNT(*) FROM actors WHERE date_of_birth>'1970-01-01';
8. SELECT MIN(domestic_takings) AS "Lowest_takings", MAX(domestic_takings) AS "Highest_takings" FROM movie_revenues;
9. SELECT SUM(movie_length) FROM movies WHERE age_certificate='15';
10. SELECT COUNT(*) FROM directors WHERE nationality='Japanese';
11. SELECT AVG(movie_length) FROM movies WHERE movie_lang='Chinese';  
12. SELECT COUNT(*), nationality FROM directors GROUP BY nationality;
13. SELECT SUM(movie_length), age_certificate, movie_lang FROM movies GROUP BY age_certificate, movie_lang ORDER BY movie_lang, age_certificate;
14. SELECT movie_lang, SUM(movie_length) FROM movies GROUP BY movie_lang HAVING SUM(movie_length)>500;
15. SELECT first_name, last_name, movie_name, release_date FROM directors JOIN movies ON directors.director_id = movies.director_id WHERE movie_lang IN ('Chinese', 'Korean','Japanese');
16. SELECT movie_name, release_date, international_takings FROM movies JOIN movie_revenues ON movies.movie_id = movie_revenues.movie_id WHERE movie_lang='English';
17. SELECT movie_name, domestic_takings, international_takings FROM movies JOIN movie_revenues ON movies.movie_id = movie_revenues.movie_id WHERE domestic_takings IS NULL OR international_takings IS NULL ORDER BY movie_name;
18. SELECT first_name, last_name, movie_name, age_certificate FROM directors LEFT JOIN movies ON directors.director_id = movies.movie_id WHERE nationality='British';
19. SELECT COUNT(*) from movies GROUP BY director_id;
20. SELECT first_name, last_name, date_of_birth FROM directors UNION SELECT first_name, last_name, date_of_birth FROM actors ORDER BY date_of_birth;
21. SELECT first_name, last_name FROM directors WHERE date_of_birth BETWEEN '1960-01-01' AND '1969-12-31' UNION SELECT first_name, last_name FROM actors WHERE date_of_birth BETWEEN '1960-01-01' AND '1969-12-31' ORDER BY last_name;
